muSEC annotated examples: ET partitioning, soil moisture dynamics and profile visualization

Overview

This vignette demonstrates the muSEC model in both mono-layer and multi-layer configurations.
It also introduces the plotting utilities plot_profile_stairs() and plot_daily_profiles() for visualizing vertical soil moisture dynamics.

We cover:

  • Hourly vs. daily forcing, with automatic daily → hourly expansion.
  • ET partitioning via LAI:
    \(E = ET_0 e^{-k_{ext} LAI}\), \(T = ET_0 - E\), with \(k_{ext}=0.378\) and default \(LAI=1\).
  • SEC evaporation, drainage (exponential or power-law), and Feddes transpiration stress.
  • Daily-mean outputs and vertical profile visualization.

Units
ET: mm per time step (mm/h or mm/day).
Ks: m/h. If you have Ks in m/day, divide by 24.
n = 1/β: the exponential shape parameter; smaller n (larger β) → faster decline.

1) Mono-layer example (constant hourly ET)

We simulate 5 days with constant \(ET_0 = 2\) mm/day (≈ 0.083 mm/h).
The model uses the default \(LAI = 1\) and \(k_{ext} = 0.378\).

dz <- 0.10
H <- 24 * 5
ET <- rep(2 / 24, H) # mm/h

res_mono <- musec_simulate(
  theta_init = 0.30,
  structure  = list(dz = dz),
  Ks = 0.5 / 24, n = 3,
  theta_s = 0.40, theta_r = 0.05,
  theta_fc = 0.20, theta_wp = 0.02, theta_star = 0.30,
  ET = ET, timestep = "hourly",
  exp_drain = TRUE, hourly_out = TRUE
)

# Plot soil moisture (hourly + daily)
plot(res_mono$series$theta, type = "l", col = 1,
     xlab = "Hour", ylab = expression(theta),
     main = "Mono-layer muSEC (constant ET)")
points(seq(24, length(res_mono$series$theta), by = 24),
       res_mono$series$theta_daily, pch = 19, col = 2)
legend("topright", legend = c("Hourly", "Daily mean"),
       lty = c(1, NA), pch = c(NA, 19), col = c(1, 2), bty = "n")


# Plot E, T, and deep drainage per step
plot(res_mono$series$evap_mm, type = "l", col = 2, lwd = 2,
     xlab = "Hour", ylab = "mm/step", main = "E, T, Deep Drainage")
lines(res_mono$series$transp_mm, col = 4, lwd = 2)
lines(res_mono$series$drain_bottom_mm, col = 1, lty = 2)
legend("topright", legend = c("E", "T", "Deep Drainage"),
       col = c(2, 4, 1), lty = c(1, 1, 2), lwd = 2, bty = "n")

2) Multi-layer example (variable ET and LAI)

We simulate 10 days over 0–60 cm soil depth, using three layers (10 cm each).
ET0 varies daily and LAI gradually increases.

S <- compute_dz_depths(Z_t = 0.6, dz_top = 0.1, dz_rest = 0.1, n_top_layers = 1)
theta_init <- infer_profile_from_averages(S$depths, S$dz, c("0.15" = 0.3, "0.60" = 0.3))

H <- 24 * 10
ET_daily <- 3 + 1.5 * sin(seq(0, 2 * pi, length.out = 10))
ET <- expand_daily_to_hourly_sinusoidale(ET_daily)
LAI <- seq(0.5, 2.0, length.out = H)

res_multi <- musec_simulate(
  theta_init, list(depths = S$depths, dz = S$dz),
  Ks = 0.5 / 24, n = 1/8,
  theta_s = 0.43, theta_r = 0,
  theta_fc = 0.13, theta_wp = 0.002, theta_star = 0.11,
  ET = ET, LAI = LAI, timestep = "hourly",
  exp_drain = TRUE, hourly_out = TRUE
)
#> 
#>  entrati in due con rate  0.001253234
#>  entrati in due con E2_0  3.366005e-05

# Plot θ by layer
matplot(t(res_multi$series$theta_mm), type = "l", lty = 1,
        xlab = "Hour", ylab = expression(theta),
        main = "Multi-layer profile (variable ET and LAI)")
legend("topright", paste("z =", round(S$depths, 2), "m"),
       lty = 1, col = 1:nrow(res_multi$series$theta_mm), cex = 0.8, bty = "n")


# Daily 0–30 cm average
theta_avg_30 <- res_multi$series$theta_avg_30
n_days <- floor(length(theta_avg_30) / 24)
theta30_daily <- tapply(theta_avg_30, rep(1:n_days, each = 24), mean)
plot(theta30_daily, type = "b", pch = 19, xlab = "Day",
     ylab = expression(bar(theta)[0-0.3]),
     main = "Daily mean depth-averaged θ (0–30 cm)")

3) Sensitivity to the extinction coefficient k

Lower k means less canopy shading, i.e., higher soil evaporation.

run_k <- function(kval) {
  musec_simulate(
    theta_init = 0.30, list(dz = 0.10),
    Ks = 0.5 / 24, n = 3,
    theta_s = 0.40, theta_r = 0.05, theta_fc = 0.25, theta_wp = 0.10, theta_star = 0.30,
    ET = rep(3, 10), k_ext = kval, timestep = "daily", hourly_out = TRUE
  )
}

res_k03 <- run_k(0.30)
res_k038 <- run_k(0.378)
res_k05 <- run_k(0.50)

plot(res_k03$series$theta, type = "l", xlab = "Hour", ylab = expression(theta),
     main = "Sensitivity to k (constant LAI=1)")
lines(res_k038$series$theta, col = 2, lty = 2)
lines(res_k05$series$theta, col = 4, lty = 3)
legend("topright", c("k=0.30", "k=0.378", "k=0.50"),
       lty = 1:3, col = c(1, 2, 4), bty = "n")

4) Profile visualization and daily wrappers

We can visualize layer-wise θ evolution using plot_profile_stairs() and automate daily comparisons with plot_daily_profiles().

# Synthetic rain and ET0 for demonstration
res_multi_plot <- res_multi
for (i in 1:24) # Add first day
  res_multi_plot$series$theta_mm <- cbind(theta_init, res_multi_plot$series$theta_mm) # paste initial condition

steps <- ncol(res_multi_plot$series$theta_mm)

rain_vec <- rbinom(steps, 1, 0.15) * runif(steps, 0, 0)
ET0_vec  <- rep(3/24, steps)

# Example: day-by-day profiles
depths <- S$depths
dz <- S$dz
plot_daily_profiles(
  res_multi_plot, depths, dz,
  rain = rain_vec, ET0 = ET0_vec,
  file = NULL, days = 0:10, xlim = c(0, 0.35),
  style = "rect", add_areas = TRUE, alpha = 0.3,
  main_prefix = "muSEC example 0–30 cm"
)
